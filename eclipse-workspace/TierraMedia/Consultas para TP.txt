
PORCENTUAL: 


SELECT promocion.nombre, tipo_promo.nombre AS 'Promocion', porcentuales.descuento AS 'Porcentaje', atraccion.nombre 
FROM promocion
JOIN porcentuales ON promocion.tipo_promo_id = 3 AND promocion.id = porcentuales.promo_id
JOIN atraccion_promocion ON promocion.id = atraccion_promocion.promocion_id
JOIN atraccion ON atraccion_promocion.atraccion_id = atraccion.id
JOIN tipo_promo ON promocion.tipo_promo_id = tipo_promo.id
ORDER BY promocion.id


ABSOLUTA:

SELECT promocion.nombre, tipo_promo.nombre AS 'Promocion', absoluta.total AS 'Costo total', atraccion.nombre 
FROM promocion
JOIN absoluta ON promocion.tipo_promo_id = 1 AND promocion.id = absoluta.promo_id
JOIN atraccion_promocion ON promocion.id = atraccion_promocion.promocion_id
JOIN atraccion ON atraccion_promocion.atraccion_id = atraccion.id
JOIN tipo_promo ON promocion.tipo_promo_id = tipo_promo.id

AXB:

SELECT promocion.nombre, tipo_promo.nombre AS 'Promocion', axb.atraccion_gratis_id AS 'Atraccion Gratis', atraccion.nombre

JOIN atraccion_promocion ON promocion.id = atraccion_promocion.promocion_id
JOIN atraccion ON atraccion_promocion.atraccion_id = atraccion.id
JOIN tipo_promo ON promocion.tipo_promo_id = tipo_promo.id
